## 概述

实现TMS NL-Ops演示系统的订单管理界面，包括订单列表、订单创建/编辑表单和订单详情页面。这些界面将作为传统UI操作的主要入口，展示系统的完整功能。

## 目标

- 创建响应式的订单列表页面，支持分页、搜索和筛选
- 实现订单创建和编辑表单，包含完整的字段验证
- 开发订单详情页面，显示订单的完整信息和相关操作
- 集成与后端API的数据交互
- 实现订单状态变更和流程管理
- 确保界面在传统UI模式下完全可用

## 技术栈

- **框架**: Next.js 15+ App Router
- **语言**: TypeScript
- **样式**: Tailwind CSS + shadcn/ui v2
- **表单**: React Hook Form + Zod validation
- **状态管理**: React Context / Zustand
- **数据获取**: SWR / TanStack Query

'use client'

import { useState, useMemo } from 'react'
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Badge } from '@/components/ui/badge'
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '@/components/ui/table'
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select'
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle, DialogTrigger } from '@/components/ui/dialog'
import { format } from 'date-fns'
import { zhCN } from 'date-fns/locale'
import { Order, OrderStatus } from '@/types/order'
import { OrderForm } from './order-form'
import { OrderDetail } from './order-detail'
import { cn } from '@/lib/utils'

interface OrderListProps {
  orders: Order[]
  onOrderCreate: (order: Partial<Order>) => Promise<void>
  onOrderUpdate: (id: string, order: Partial<Order>) => Promise<void>
  onOrderDelete: (id: string) => Promise<void>
}

export function OrderList({ orders, onOrderCreate, onOrderUpdate, onOrderDelete }: OrderListProps) {
  const [searchTerm, setSearchTerm] = useState('')
  const [statusFilter, setStatusFilter] = useState<string>('all')
  const [selectedOrder, setSelectedOrder] = useState<Order | null>(null)
  const [showCreateDialog, setShowCreateDialog] = useState(false)
  const [showDetailDialog, setShowDetailDialog] = useState(false)

  // 筛选和搜索订单
  const filteredOrders = useMemo(() => {
    return orders.filter(order => {
      const matchesSearch = order.orderNumber.toLowerCase().includes(searchTerm.toLowerCase()) ||
        order.consigneeName.toLowerCase().includes(searchTerm.toLowerCase()) ||
        order.consignorName.toLowerCase().includes(searchTerm.toLowerCase())

      const matchesStatus = statusFilter === 'all' || order.status === statusFilter

      return matchesSearch && matchesStatus
    })
  }, [orders, searchTerm, statusFilter])

  // 获取状态对应的样式
  const getStatusVariant = (status: OrderStatus) => {
    switch (status) {
      case 'PENDING':
        return 'secondary'
      case 'ACCEPTED':
        return 'default'
      case 'IN_TRANSIT':
        return 'default'
      case 'DELIVERED':
        return 'default'
      case 'COMPLETED':
        return 'default'
      case 'CANCELLED':
        return 'destructive'
      default:
        return 'secondary'
    }
  }

  // 获取状态显示文本
  const getStatusText = (status: OrderStatus) => {
    switch (status) {
      case 'PENDING': return '待处理'
      case 'ACCEPTED': return '已接受'
      case 'IN_TRANSIT': return '运输中'
      case 'DELIVERED': return '已送达'
      case 'COMPLETED': return '已完成'
      case 'CANCELLED': return '已取消'
      default: return status
    }
  }

  // 格式化重量
  const formatWeight = (weight: number) => {
    return `${weight} kg`
  }

  // 格式化体积
  const formatVolume = (volume: number) => {
    return `${volume} m³`
  }

  // 处理订单选择
  const handleOrderSelect = (order: Order) => {
    setSelectedOrder(order)
    setShowDetailDialog(true)
  }

  // 处理订单创建
  const handleOrderCreate = async (orderData: Partial<Order>) => {
    await onOrderCreate(orderData)
    setShowCreateDialog(false)
  }

  // 处理订单更新
  const handleOrderUpdate = async (orderId: string, orderData: Partial<Order>) => {
    await onOrderUpdate(orderId, orderData)
    setShowDetailDialog(false)
    setSelectedOrder(null)
  }

  // 处理订单删除
  const handleOrderDelete = async (orderId: string) => {
    if (window.confirm('确定要删除这个订单吗？')) {
      await onOrderDelete(orderId)
      setShowDetailDialog(false)
      setSelectedOrder(null)
    }
  }

  return (
    <div className="space-y-6">
      {/* 页面标题和操作栏 */}
      <div className="flex justify-between items-center">
        <div>
          <h1 className="text-3xl font-bold tracking-tight">订单管理</h1>
          <p className="text-muted-foreground">管理所有运输订单</p>
        </div>

        <Dialog open={showCreateDialog} onOpenChange={setShowCreateDialog}>
          <DialogTrigger asChild>
            <Button>
              创建订单
            </Button>
          </DialogTrigger>
          <DialogContent className="max-w-4xl max-h-[90vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle>创建新订单</DialogTitle>
              <DialogDescription>
                填写订单信息创建新的运输订单
              </DialogDescription>
            </DialogHeader>
            <OrderForm
              onSubmit={handleOrderCreate}
              onCancel={() => setShowCreateDialog(false)}
            />
          </DialogContent>
        </Dialog>
      </div>

      {/* 搜索和筛选栏 */}
      <Card>
        <CardContent className="pt-6">
          <div className="flex space-x-4">
            <div className="flex-1">
              <Input
                placeholder="搜索订单号、发货人或收货人..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="max-w-sm"
              />
            </div>
            <Select value={statusFilter} onValueChange={setStatusFilter}>
              <SelectTrigger className="w-[180px]">
                <SelectValue placeholder="筛选状态" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">所有状态</SelectItem>
                <SelectItem value="PENDING">待处理</SelectItem>
                <SelectItem value="ACCEPTED">已接受</SelectItem>
                <SelectItem value="IN_TRANSIT">运输中</SelectItem>
                <SelectItem value="DELIVERED">已送达</SelectItem>
                <SelectItem value="COMPLETED">已完成</SelectItem>
                <SelectItem value="CANCELLED">已取消</SelectItem>
              </SelectContent>
            </Select>
          </div>
        </CardContent>
      </Card>

      {/* 订单列表 */}
      <Card>
        <CardHeader>
          <CardTitle>订单列表</CardTitle>
          <CardDescription>
            共找到 {filteredOrders.length} 个订单
          </CardDescription>
        </CardHeader>
        <CardContent>
          <Table>
            <TableHeader>
              <TableRow>
                <TableHead>订单号</TableHead>
                <TableHead>发货人</TableHead>
                <TableHead>收货人</TableHead>
                <TableHead>重量</TableHead>
                <TableHead>体积</TableHead>
                <TableHead>状态</TableHead>
                <TableHead>创建时间</TableHead>
                <TableHead className="text-right">操作</TableHead>
              </TableRow>
            </TableHeader>
            <TableBody>
              {filteredOrders.map((order) => (
                <TableRow
                  key={order.id}
                  className="cursor-pointer hover:bg-muted/50"
                  onClick={() => handleOrderSelect(order)}
                >
                  <TableCell className="font-medium">{order.orderNumber}</TableCell>
                  <TableCell>{order.consignorName}</TableCell>
                  <TableCell>{order.consigneeName}</TableCell>
                  <TableCell>{formatWeight(order.weight)}</TableCell>
                  <TableCell>{formatVolume(order.volume)}</TableCell>
                  <TableCell>
                    <Badge variant={getStatusVariant(order.status)}>
                      {getStatusText(order.status)}
                    </Badge>
                  </TableCell>
                  <TableCell>
                    {format(new Date(order.createdAt), 'yyyy-MM-dd HH:mm', { locale: zhCN })}
                  </TableCell>
                  <TableCell className="text-right">
                    <Button
                      variant="ghost"
                      size="sm"
                      onClick={(e) => {
                        e.stopPropagation()
                        handleOrderSelect(order)
                      }}

                      查看
                    </Button>
                  </TableCell>
                </TableRow>
              ))}
            </TableBody>
          </Table>
        </CardContent>

const orderFormSchema = z.object({
  orderType: z.enum(['EXPRESS', 'STANDARD', 'HEAVY']),
  consignorName: z.string().min(1, '发货人姓名不能为空'),
  consignorPhone: z.string().min(1, '发货人电话不能为空'),
  consignorAddress: z.string().min(1, '发货地址不能为空'),
  consigneeName: z.string().min(1, '收货人姓名不能为空'),
  consigneePhone: z.string().min(1, '收货人电话不能为空'),
  consigneeAddress: z.string().min(1, '收货地址不能为空'),
  goodsName: z.string().min(1, '货物名称不能为空'),
  weight: z.number().min(0.1, '重量必须大于0'),
  volume: z.number().min(0.001, '体积必须大于0'),
  specialRequirements: z.string().optional(),
  deliveryTime: z.string().optional(),
  pickupTime: z.string().optional(),
  estimatedPrice: z.number().min(0, '预估价格不能为负数').optional(),
})

type OrderFormData = z.infer<typeof orderFormSchema>

interface OrderFormProps {
  order?: Order
  onSubmit: (data: Partial<Order>) => Promise<void>
  onCancel: () => void
}

export function OrderForm({ order, onSubmit, onCancel }: OrderFormProps) {
  const [isSubmitting, setIsSubmitting] = useState(false)

  const form = useForm<OrderFormData>({
    resolver: zodResolver(orderFormSchema),
    defaultValues: {
      orderType: order?.orderType || 'STANDARD',
      consignorName: order?.consignorName || '',
      consignorPhone: order?.consignorPhone || '',
      consignorAddress: order?.consignorAddress || '',
      consigneeName: order?.consigneeName || '',
      consigneePhone: order?.consigneePhone || '',
      consigneeAddress: order?.consigneeAddress || '',
      goodsName: order?.goodsName || '',
      weight: order?.weight || 0,
      volume: order?.volume || 0,
      specialRequirements: order?.specialRequirements || '',
      deliveryTime: order?.deliveryTime ? new Date(order.deliveryTime).toISOString().slice(0, 16) : '',
      pickupTime: order?.pickupTime ? new Date(order.pickupTime).toISOString().slice(0, 16) : '',
      estimatedPrice: order?.estimatedPrice || 0,
    },
  })

  const handleSubmit = async (data: OrderFormData) => {
    setIsSubmitting(true)
    try {
      const orderData: Partial<Order> = {
        ...data,
        pickupTime: data.pickupTime ? new Date(data.pickupTime) : undefined,
        deliveryTime: data.deliveryTime ? new Date(data.deliveryTime) : undefined,
      }

      await onSubmit(orderData)
    } catch (error) {
      console.error('提交订单失败:', error)
    } finally {
      setIsSubmitting(false)
    }
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(handleSubmit)} className="space-y-6">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
          {/* 订单基本信息 */}
          <Card>
            <CardHeader>
              <CardTitle>基本信息</CardTitle>
              <CardDescription>订单的基本信息</CardDescription>
            </CardHeader>
            <CardContent className="space-y-4">
              <FormField
                control={form.control}
                name="orderType"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>订单类型</FormLabel>
                    <Select onValueChange={field.onChange} defaultValue={field.value}>
                      <FormControl>
                        <SelectTrigger>
                          <SelectValue placeholder="选择订单类型" />
                        </SelectTrigger>
                      </FormControl>
                      <SelectContent>
                        <SelectItem value="EXPRESS">快递</SelectItem>
                        <SelectItem value="STANDARD">标准</SelectItem>
                        <SelectItem value="HEAVY">重货</SelectItem>
                      </SelectContent>
                    </Select>
                    <FormMessage />
                  </FormItem>
                )}
              />

              <FormField
                control={form.control}
                name="goodsName"
                render={({ field }) => (
                  <FormItem>
                    <FormLabel>货物名称</FormLabel>
                    <FormControl>
                      <Input placeholder="请输入货物名称" {...field} />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}
              />

                      <FormControl>
                        <Input
                          type="number"
                          step="0.001"

      default: return type
    }
  }

  // 格式化重量
  const formatWeight = (weight: number) => {
    return `${weight} kg`
  }

  // 格式化体积
  const formatVolume = (volume: number) => {
    return `${volume} m³`
  }

  // 格式化价格
  const formatPrice = (price: number) => {
    return `¥${price.toFixed(2)}`
  }

  // 处理订单更新
  const handleOrderUpdate = async (orderData: Partial<Order>) => {
    await onUpdate(order.id, orderData)
    setShowEditForm(false)
    setIsEditing(false)
  }

  // 处理订单删除
  const handleOrderDelete = async () => {
    await onDelete(order.id)
  }

  // 处理状态变更
  const handleStatusChange = async (newStatus: OrderStatus) => {
    await onUpdate(order.id, { status: newStatus })
  }

  return (
    <div className="space-y-6">
      {/* 订单基本信息 */}
      <div className="flex justify-between items-start">
        <div>
          <h2 className="text-2xl font-bold">订单 {order.orderNumber}</h2>
          <div className="flex items-center space-x-2 mt-2">
            <Badge variant={getStatusVariant(order.status)}>
              {getStatusText(order.status)}
            </Badge>
            <Badge variant="outline">
              {getOrderTypeText(order.orderType)}
            </Badge>
          </div>
        </div>

        <div className="flex space-x-2">
          {!isEditing && (
            <Button variant="outline" onClick={() => setShowEditForm(true)}>
              编辑
            </Button>
          )}
          {order.status === 'PENDING' && (
            <Button onClick={() => handleStatusChange('ACCEPTED')}>
              接受订单
            </Button>
          )}
          {order.status === 'ACCEPTED' && (
            <Button onClick={() => handleStatusChange('IN_TRANSIT')}>
              开始运输
            </Button>
          )}
          {order.status === 'IN_TRANSIT' && (
            <Button onClick={() => handleStatusChange('DELIVERED')}>
              标记送达
            </Button>
          )}
          {order.status === 'DELIVERED' && (
            <Button onClick={() => handleStatusChange('COMPLETED')}>
              完成订单
            </Button>
          )}
          {(order.status === 'PENDING' || order.status === 'ACCEPTED') && (
            <Button variant="destructive" onClick={() => handleStatusChange('CANCELLED')}>
              取消订单
            </Button>
          )}
        </div>
      </div>

      {showEditForm ? (
        <OrderForm
          order={order}
          onSubmit={handleOrderUpdate}
          onCancel={() => setShowEditForm(false)}
        />
      ) : (
        <Tabs defaultValue="basic" className="w-full">
          <TabsList className="grid w-full grid-cols-4">
            <TabsTrigger value="basic">基本信息</TabsTrigger>
            <TabsTrigger value="contacts">联系信息</TabsTrigger>
            <TabsTrigger value="timeline">时间线</TabsTrigger>
            <TabsTrigger value="documents">相关文档</TabsTrigger>
          </TabsList>

          <TabsContent value="basic">
            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              <Card>
                <CardHeader>
                  <CardTitle>货物信息</CardTitle>
                </CardHeader>
                <CardContent className="space-y-4">
                  <div className="flex justify-between">
                    <span className="text-muted-foreground">货物名称:</span>
                    <span className="font-medium">{order.goodsName}</span>
                  </div>

                  {order.specialRequirements && (

                  <div className="flex justify-between">

                        {format(new Date(order.pickupTime), 'yyyy-MM-dd HH:mm:ss', { locale: zhCN })}

                  {order.deliveryTime && (
                    <div className="flex justify-between">
                      <span className="text-muted-foreground">送达时间:</span>
                      <span className="font-medium">
                        {format(new Date(order.deliveryTime), 'yyyy-MM-dd HH:mm:ss', { locale: zhCN })}

                    </div>

                </CardContent>
              </Card>
            </div>
          </TabsContent>

          <TabsContent value="contacts">
            <div className="grid grid-cols-1 md:grid-cols-2 gap-6">
              <Card>
                <CardHeader>
                  <CardTitle>发货人信息</CardTitle>
                </CardHeader>
                <CardContent className="space-y-4">
                  <div className="flex justify-between">
                    <span className="text-muted-foreground">姓名:</span>

                        {format(new Date(order.createdAt), 'yyyy-MM-dd HH:mm:ss', { locale: zhCN })}
                      </p>
                    </div>
                  </div>

                  {order.status !== 'PENDING' && (
                    <div className="flex items-center space-x-4">

                      <div className="w-4 h-4 bg-green-600 rounded-full"></div>

                        </p>
                      </div>
                    </div>
                  ) : null}

                  {order.status === 'CANCELLED' && (
                    <div className="flex items-center space-x-4">
                      <div className="w-4 h-4 bg-red-500 rounded-full"></div>
                      <div>
                        <p className="font-medium">订单取消</p>
                        <p className="text-sm text-muted-foreground">订单已被取消</p>
                      </div>
                    </div>
                  )}
                </div>
              </CardContent>
            </Card>
          </TabsContent>

          <TabsContent value="documents">
            <Card>
              <CardHeader>
                <CardTitle>相关文档</CardTitle>
                <CardDescription>订单相关的文档和证明</CardDescription>
              </CardHeader>
              <CardContent>
                <p className="text-muted-foreground">暂无相关文档</p>
              </CardContent>
            </Card>
          </TabsContent>
        </Tabs>
      )}

      {/* 底部操作按钮 */}
      {!showEditForm && (
        <div className="flex justify-between">
          <Button variant="outline" onClick={onClose}>
            关闭
          </Button>
          <Button variant="destructive" onClick={handleOrderDelete}>
            删除订单
          </Button>
        </div>
      )}
    </div>
  )
}
