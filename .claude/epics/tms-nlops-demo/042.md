// scripts/generate-api-docs.ts
import { writeFileSync, mkdirSync, existsSync } from 'fs';
import { join } from 'path';
import { parse } from 'yaml';

interface ApiEndpoint {
  path: string;
  method: string;
  summary: string;
  description: string;
  parameters?: any[];
  requestBody?: any;
  responses: any;
  tags: string[];
}

class ApiDocsGenerator {
  private outputPath: string;

  constructor(outputPath: string = './docs/api') {
    this.outputPath = outputPath;
    this.ensureOutputDir();
  }

  private ensureOutputDir() {
    if (!existsSync(this.outputPath)) {
      mkdirSync(this.outputPath, { recursive: true });
    }
  }

  async generateFromOpenApi(specPath: string) {
    const specContent = await this.readFile(specPath);
    const openApiSpec = parse(specContent);

    // 生成索引
    await this.generateIndex(openApiSpec);

    // 生成各API文档
    for (const [path, pathItem] of Object.entries(openApiSpec.paths)) {
      for (const [method, operation] of Object.entries(pathItem as any)) {
        if (method === 'parameters') continue;

        const endpoint: ApiEndpoint = {
          path,
          method: method.toUpperCase(),
          summary: operation.summary || '',
          description: operation.description || '',
          parameters: operation.parameters,
          requestBody: operation.requestBody,
          responses: operation.responses,
          tags: operation.tags || []
        };

        await this.generateEndpointDoc(endpoint);
      }
    }

    // 生成分类文档
    await this.generateTagDocs(openApiSpec);

    console.log('API文档生成完成!');
  }

  private async generateIndex(openApiSpec: any) {
    const indexContent = `# TMS NL-Ops API 文档

## 概述

TMS NL-Ops 演示系统提供了一套完整的RESTful API，支持传统的运输管理功能和AI驱动的智能交互。

### 基础信息

- **Base URL**: \`https://api.tms-demo.example.com\`
- **版本**: v1
- **认证**: Bearer Token
- **内容类型**: application/json

### API 概览

${Object.entries(openApiSpec.paths).map(([path, pathItem]) => {
  return Object.entries(pathItem as any)
    .filter(([method]) => method !== 'parameters')
    .map(([method, operation]: [string, any]) => {
      const badgeColor = this.getMethodBadgeColor(method.toUpperCase());
      return `- [\`${method.toUpperCase()} ${path}\`](${method.toUpperCase()}_${path.replace(/\//g, '_')}.md) - ${operation.summary}`;
    })
    .join('\n');
}).join('\n')}

### 认证

所有API请求都需要在Header中包含认证token：

\`\`\`http
Authorization: Bearer your-api-token
Content-Type: application/json
\`\`\`

### 响应格式

所有API响应都采用统一的JSON格式：

\`\`\`json
{
  "success": true,
  "data": {},
  "message": "操作成功",
  "timestamp": "2024-01-01T00:00:00Z"
}
\`\`\`

### 错误处理

错误响应格式：

\`\`\`json
{
  "success": false,
  "error": {
    "code": "ERROR_CODE",
    "message": "错误描述",
    "details": {}
  },
  "timestamp": "2024-01-01T00:00:00Z"
}
\`\`\`

## 快速开始

### 1. 获取API Token

请联系管理员获取API访问令牌。

### 2. 发送第一个请求

\`\`\`bash
curl -X GET "https://api.tms-demo.example.com/api/tms/orders" \\
  -H "Authorization: Bearer your-token" \\
  -H "Content-Type: application/json"
\`\`\`

### 3. 查看响应

系统将返回订单列表数据。

---

*文档生成时间: ${new Date().toISOString()}*
`;

    writeFileSync(join(this.outputPath, 'README.md'), indexContent);
  }

  private async generateEndpointDoc(endpoint: ApiEndpoint) {
    const fileName = `${endpoint.method}_${endpoint.path.replace(/\//g, '_')}.md`;
    const filePath = join(this.outputPath, fileName);

    const content = `# ${endpoint.method} ${endpoint.path}

${endpoint.description}

## 基本信息

- **方法**: \`${endpoint.method}\`
- **路径**: \`${endpoint.path}\`
- **标签**: ${endpoint.tags.map(tag => `\`${tag}\``).join(', ')}

## 请求参数

${endpoint.parameters ? this.generateParametersTable(endpoint.parameters) : '无参数'}

${endpoint.requestBody ? this.generateRequestBodySection(endpoint.requestBody) : ''}

## 响应

${this.generateResponsesSection(endpoint.responses)}

## 代码示例

### JavaScript

\`\`\`javascript
const response = await fetch('${endpoint.path}', {
  method: '${endpoint.method}',
  headers: {
    'Authorization': 'Bearer your-token',
    'Content-Type': 'application/json'
  }${endpoint.requestBody ? `,
  body: JSON.stringify({
    // 请求数据
  })` : ''}
});

const data = await response.json();
console.log(data);
\`\`\`

### Python

\`\`\`python
import requests

url = '${endpoint.path}'
headers = {
    'Authorization': 'Bearer your-token',
    'Content-Type': 'application/json'
}
${endpoint.requestBody ? `data = {
    # 请求数据
}` : ''}

response = requests.${endpoint.method.toLowerCase()}(url, headers=headers${endpoint.requestBody ? ', json=data' : ''})
print(response.json())
\`\`\`

### cURL

\`\`\`bash
curl -X ${endpoint.method} "${endpoint.path}" \\
  -H "Authorization: Bearer your-token" \\
  -H "Content-Type: application/json"${endpoint.requestBody ? ` \\
  -d '{
    # 请求数据
  }'` : ''}
\`\`\`

## 错误码

| 错误码 | 描述 |
|--------|------|
| 400 | 请求参数错误 |
| 401 | 未授权 |
| 403 | 禁止访问 |
| 404 | 资源不存在 |
| 500 | 服务器内部错误 |

---

*文档生成时间: ${new Date().toISOString()}*
`;

    writeFileSync(filePath, content);
  }

  private generateParametersTable(parameters: any[]) {
    const table = `| 参数名 | 类型 | 必填 | 描述 | 示例 |
|--------|------|------|------|------|
${parameters.map(param => {
  const required = param.required ? '是' : '否';
  const example = param.example || '';
  return `| ${param.name} | ${param.schema?.type || 'string'} | ${required} | ${param.description || ''} | ${example} |`;
}).join('\n')}`;

    return table;
  }

  private generateRequestBodySection(requestBody: any) {
    const schema = requestBody.content['application/json'].schema;
    return `## 请求体

\`\`\`json
${JSON.stringify(schema.example || {}, null, 2)}
\`\`\`

### 字段说明

${this.generateSchemaFields(schema.properties || schema.items?.properties || {})}`;
  }

  private generateSchemaFields(properties: any, prefix = '') {
    return Object.entries(properties).map(([key, prop]) => {
      const type = prop.type;
      const description = prop.description || '';
      const required = prop.required ? '（必填）' : '';
      return `- \`${prefix}${key}\`: ${type}${required} - ${description}`;
    }).join('\n');
  }

  private generateResponsesSection(responses: any) {
    return Object.entries(responses)
      .map(([status, response]: [string, any]) => {
        const schema = response.content?.['application/json']?.schema;
        return `### ${status} ${response.description || ''}

${schema ? `\`\`\`json
${JSON.stringify(schema.example || {}, null, 2)}
\`\`\`` : ''}`;
      })
      .join('\n');
  }

  private getMethodBadgeColor(method: string) {
    const colors = {
      GET: 'green',
      POST: 'blue',
      PUT: 'orange',
      DELETE: 'red',
      PATCH: 'purple'
    };
    return colors[method as keyof typeof colors] || 'gray';
  }

  private async readFile(path: string): Promise<string> {
    const fs = require('fs');
    return fs.readFileSync(path, 'utf8');
  }

  private async generateTagDocs(openApiSpec: any) {
    const tags = openApiSpec.tags || [];

    for (const tag of tags) {
      const tagEndpoints = Object.entries(openApiSpec.paths)
        .flatMap(([path, pathItem]) =>
          Object.entries(pathItem as any)
            .filter(([method]) => method !== 'parameters')
            .map(([method, operation]: [string, any]) => ({
              path,
              method: method.toUpperCase(),
              operation
            }))
        )
        .filter(({ operation }) => operation.tags?.includes(tag.name));

      const tagContent = `# ${tag.name}

${tag.description || ''}

## 相关API

${tagEndpoints.map(({ path, method, operation }) => {
  return `- [\`${method} ${path}\`](${method.toUpperCase()}_${path.replace(/\//g, '_')}.md) - ${operation.summary}`;
}).join('\n')}

---

*文档生成时间: ${new Date().toISOString()}*
`;

      writeFileSync(join(this.outputPath, `${tag.name}.md`), tagContent);
    }
  }
}

// 使用示例
const generator = new ApiDocsGenerator();
generator.generateFromOpenApi('./openapi.yaml');
